* Installing docker inside docker is not recommended. Instead mount the host machine's Docker socket in the container
* by using below commands. Credit goes to getintodevops

docker run -p 8080:8085 -v /var/run/docker.sock:/var/run/docker.sock --name jenkins getintodevops/jenkins-withdocker:lts

docker exec -it -u root jenkins bash

apt-get update && \
apt-get -y install apt-transport-https \
     ca-certificates \
     curl \
     gnupg2 \
     software-properties-common && \
curl -fsSL https://download.docker.com/linux/$(. /etc/os-release; echo "$ID")/gpg > /tmp/dkey; apt-key add /tmp/dkey && \
add-apt-repository \
   "deb [arch=amd64] https://download.docker.com/linux/$(. /etc/os-release; echo "$ID") \
   $(lsb_release -cs) \
   stable" && \
apt-get update && \
apt-get -y install docker-ce

docker exec --user root -it jenkins bash
sudo service jenkins restart

docker ps

docker exec jenkins cat /var/jenkins_home/secrets/initialAdminPassword
